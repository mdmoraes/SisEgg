SELECT CLI_CLIENTE, CLI_NFANTASIA,CLI_ENDERECO,
CLI_ENDNUMERO,
CLI_BAIRRO, CLI_CIDADE, CLI_ESTADO, CLI_CEP,
CLI_FONE1, CLI_CELULAR,
CLI_CGC, CLI_INSCRICAO, CLI_CPF, CLI_RG,
CLI_SEGUNDA, CLI_TERCA, CLI_QUARTA,
CLI_QUINTA, CLI_SEXTA, CLI_SABADO,
Cli_RotaSegunda, Cli_RotaTerca, Cli_RotaQuarta,
Cli_RotaQuinta, Cli_RotaSexta, Cli_RotaSabado
FROM CLIENTES
WHERE CLI_SEGUNDA like (:VarDia) or CLI_TERCA like (:VarDia) or
      CLI_QUARTA like (:VarDia) OR CLI_QUINTA like (:VarDia) OR
      CLI_SEXTA like (:VarDia) OR CLI_SABADO like (:VarDia)
ORDER BY Cli_RotaSegunda
